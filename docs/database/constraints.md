# SahakariMS — Database: Constraints

## Overview

Database constraints ensure data integrity at the storage layer. Even if application validation fails, the database acts as the final safety net.

---

## Primary Keys

All tables use UUID primary keys generated by PostgreSQL:

```sql
-- Standard pattern used across all tables
id UUID PRIMARY KEY DEFAULT gen_random_uuid()
```

---

## Unique Constraints

```sql
-- Members
ALTER TABLE members ADD CONSTRAINT uq_members_citizenship
    UNIQUE (citizenship_number) WHERE citizenship_number IS NOT NULL AND is_deleted = FALSE;

ALTER TABLE members ADD CONSTRAINT uq_members_member_code
    UNIQUE (member_code);

-- Users
ALTER TABLE users ADD CONSTRAINT uq_users_username
    UNIQUE (username) WHERE is_deleted = FALSE;

ALTER TABLE users ADD CONSTRAINT uq_users_email
    UNIQUE (email) WHERE is_deleted = FALSE;

-- Branches
ALTER TABLE branches ADD CONSTRAINT uq_branches_code
    UNIQUE (branch_code);

-- Account numbers
ALTER TABLE saving_accounts ADD CONSTRAINT uq_saving_accounts_number
    UNIQUE (account_number);

ALTER TABLE loans ADD CONSTRAINT uq_loans_number
    UNIQUE (loan_number);

-- Accounting
ALTER TABLE accounting.accounts ADD CONSTRAINT uq_accounts_code_branch
    UNIQUE (branch_id, account_code);

ALTER TABLE accounting.fiscal_years ADD CONSTRAINT uq_fiscal_year_branch
    UNIQUE (branch_id, year_bs);
```

## Check Constraints

```sql
-- Financial amounts must be positive
ALTER TABLE saving_transactions ADD CONSTRAINT chk_saving_txn_positive_amount
    CHECK (amount > 0);

ALTER TABLE loan_payments ADD CONSTRAINT chk_loan_payment_positive
    CHECK (amount > 0 AND principal_paid >= 0 AND interest_paid >= 0 AND penalty_paid >= 0);

-- Interest rates in reasonable range
ALTER TABLE loan_products ADD CONSTRAINT chk_loan_product_rate
    CHECK (min_interest_rate >= 0 AND max_interest_rate <= 50);

ALTER TABLE saving_schemes ADD CONSTRAINT chk_saving_scheme_rate
    CHECK (interest_rate >= 0 AND interest_rate <= 50);

-- Loan tenure
ALTER TABLE loans ADD CONSTRAINT chk_loan_tenure
    CHECK (tenure_months > 0 AND tenure_months <= 360);

-- Shares
ALTER TABLE share_transactions ADD CONSTRAINT chk_shares_positive
    CHECK (number_of_shares > 0 AND amount_per_share > 0);

-- Member age (must be 18+)
ALTER TABLE members ADD CONSTRAINT chk_member_adult
    CHECK (date_of_birth_ad <= CURRENT_DATE - INTERVAL '18 years');

-- EMI number in schedule must be positive
ALTER TABLE loan_schedules ADD CONSTRAINT chk_emi_number_positive
    CHECK (emi_number > 0);

-- Voucher entries: double-entry balance
-- (enforced at application layer, spot-checked by nightly job)

-- Gender
ALTER TABLE members ADD CONSTRAINT chk_member_gender
    CHECK (gender IN ('Male', 'Female', 'Other'));

-- Account status transitions
ALTER TABLE saving_accounts ADD CONSTRAINT chk_account_status
    CHECK (status IN ('Active', 'Dormant', 'Frozen', 'Closed'));

-- Loan status
ALTER TABLE loans ADD CONSTRAINT chk_loan_status
    CHECK (status IN ('Pending', 'UnderReview', 'Approved', 'Rejected',
                      'Active', 'Overdue', 'Closed', 'WrittenOff'));

-- NPA classification
ALTER TABLE loans ADD CONSTRAINT chk_loan_npa
    CHECK (npa_classification IN ('Standard', 'Watchlist', 'Substandard',
                                  'Doubtful', 'Loss') OR npa_classification IS NULL);
```

## Foreign Key Constraints

```sql
-- Members → Branch
ALTER TABLE members ADD CONSTRAINT fk_members_branch
    FOREIGN KEY (branch_id) REFERENCES branches(id);

-- Saving Accounts → Member + Branch + Scheme
ALTER TABLE saving_accounts ADD CONSTRAINT fk_saving_accounts_member
    FOREIGN KEY (member_id) REFERENCES members(id);
ALTER TABLE saving_accounts ADD CONSTRAINT fk_saving_accounts_branch
    FOREIGN KEY (branch_id) REFERENCES branches(id);
ALTER TABLE saving_accounts ADD CONSTRAINT fk_saving_accounts_scheme
    FOREIGN KEY (scheme_id) REFERENCES saving_schemes(id);

-- Saving Transactions → Account
ALTER TABLE saving_transactions ADD CONSTRAINT fk_saving_txn_account
    FOREIGN KEY (account_id) REFERENCES saving_accounts(id);

-- Loans → Member + Branch + Product
ALTER TABLE loans ADD CONSTRAINT fk_loans_member
    FOREIGN KEY (member_id) REFERENCES members(id);
ALTER TABLE loans ADD CONSTRAINT fk_loans_branch
    FOREIGN KEY (branch_id) REFERENCES branches(id);
ALTER TABLE loans ADD CONSTRAINT fk_loans_product
    FOREIGN KEY (product_id) REFERENCES loan_products(id);

-- Loan Schedules → Loan
ALTER TABLE loan_schedules ADD CONSTRAINT fk_loan_schedules_loan
    FOREIGN KEY (loan_id) REFERENCES loans(id);

-- Loan Guarantors → Loan + Member (guarantor)
ALTER TABLE loan_guarantors ADD CONSTRAINT fk_guarantors_loan
    FOREIGN KEY (loan_id) REFERENCES loans(id);
ALTER TABLE loan_guarantors ADD CONSTRAINT fk_guarantors_member
    FOREIGN KEY (guarantor_member_id) REFERENCES members(id);

-- Voucher Entries → Voucher + Account
ALTER TABLE accounting.voucher_entries ADD CONSTRAINT fk_entries_voucher
    FOREIGN KEY (voucher_id) REFERENCES accounting.vouchers(id);
ALTER TABLE accounting.voucher_entries ADD CONSTRAINT fk_entries_account
    FOREIGN KEY (account_id) REFERENCES accounting.accounts(id);

-- Users → Branch
ALTER TABLE users ADD CONSTRAINT fk_users_branch
    FOREIGN KEY (branch_id) REFERENCES branches(id);
```

## Deferred Constraints

For double-entry accounting, we allow deferred checking within a transaction:

```sql
-- Voucher entries balance check (deferred until end of transaction)
ALTER TABLE accounting.vouchers ADD CONSTRAINT chk_voucher_balanced
    CHECK (
        (SELECT COALESCE(SUM(debit_amount), 0) FROM accounting.voucher_entries WHERE voucher_id = id) =
        (SELECT COALESCE(SUM(credit_amount), 0) FROM accounting.voucher_entries WHERE voucher_id = id)
    ) DEFERRABLE INITIALLY DEFERRED;
```

## Constraint Violation Handling

```csharp
// Infrastructure/Persistence/Exceptions/DatabaseExceptionHandler.cs
public class DatabaseExceptionTranslator
{
    public static DomainException Translate(PostgresException ex)
    {
        return ex.SqlState switch
        {
            "23505" => ex.ConstraintName switch  // Unique violation
            {
                "uq_members_citizenship" =>
                    new DomainException("A member with this citizenship number already exists."),
                "uq_users_username" =>
                    new DomainException("This username is already taken."),
                "uq_accounts_code_branch" =>
                    new DomainException("An account with this code already exists in this branch."),
                _ => new DomainException($"Duplicate value: {ex.ConstraintName}")
            },
            "23503" => new DomainException(  // FK violation
                $"Cannot delete record that has related data: {ex.ConstraintName}"),
            "23514" => new DomainException(  // Check violation
                $"Data validation failed: {ex.ConstraintName}"),
            _ => new DomainException($"Database error: {ex.Message}")
        };
    }
}
```
